Структура с Звездами Голивуда найти ср рост по И, вывести Ф совпавших и Ср Рост

```c

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct Person{
    char name[100];
    char surname[100];
    int height;
} Person;

int main(){

    Person Hollywood_People[1000];
    int amount;
    scanf("%d", &amount);
    for(int i=0; i < amount; i++){
        scanf("%s",Hollywood_People[i].name);
        scanf("%s",Hollywood_People[i].surname);
        scanf("%d",&Hollywood_People[i].height);
    }
    char regex_name[100];
    scanf("%s",regex_name);
    float average_height=0;
    int count=0;
    for(int i=0; i < amount; i++){
        if(!(strcmp(regex_name, Hollywood_People[i].name))){
            count++;
            average_height+=Hollywood_People[i].height;
        }
    }
    int is_first_printed=0;
    for(int i=0; i < amount; i++){
        if(!(strcmp(regex_name, Hollywood_People[i].name))){
            if(!is_first_printed){
                printf("%s", Hollywood_People[i].surname);
                is_first_printed=1;
            }
            else
                printf(", %s", Hollywood_People[i].surname);
        }
    }

  if (count != 0)
    printf(", %.f", (float)average_height/count);
  else
    printf("n/a");
    return 0;
}
```
